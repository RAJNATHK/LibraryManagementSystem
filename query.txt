DATABASE QUERIES

1. Create Author Table

CREATE TABLE Author_tbl (
    Author_code NUMBER(5) PRIMARY KEY,
    Author_name VARCHAR2(20) NOT NULL,
    Contact_no NUMBER(10)
);

3. Create Book Table

CREATE TABLE Book_tbl (
    ISBN VARCHAR2(10) PRIMARY KEY,
    Book_title VARCHAR2(20) NOT NULL,
    Book_type CHAR(1) CHECK (Book_type IN ('T','G')),
    Author_code NUMBER(5) NOT NULL,
    Book_cost NUMBER(8,2) NOT NULL,
    CONSTRAINT fk_author
    FOREIGN KEY (Author_code)
    REFERENCES Author_tbl(Author_code)
);

4. Sample Insert Into Book

INSERT INTO Book_tbl
(ISBN, Book_title, Book_type, Author_code, Book_cost)
VALUES
('101', 'Java', 'T', 1, 500);

5. View All Books

SELECT * FROM Book_tbl;

6. View Book By ISBN

SELECT * FROM Book_tbl
WHERE ISBN = '101';

7. Join Query (Book + Author)

SELECT b.ISBN,
       b.Book_title,
       b.Book_type,
       a.Author_name,
       a.Contact_no,
       b.Book_cost
FROM Book_tbl b
JOIN Author_tbl a
ON b.Author_code = a.Author_code;
